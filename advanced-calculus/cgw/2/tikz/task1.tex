\begin{tikzpicture}
  \pgfmathdeclarefunction{cbrt}{1}{%
    \pgfmathparse{sign(#1)*pow(abs(#1),1/3)}%
  }

  \begin{axis}[
        width=12cm, height=12cm,
        axis lines=middle,
        axis line style={thick, black},
        xlabel=$x$,
        ylabel=$y$,
        xlabel style={right},
        ylabel style={above},
        xmin=-1.5, xmax=1.5,
        ymin=-2.5, ymax=1.5,
        xtick={-1.5,-1,-0.5,0,0.5,1,1.5},
        ytick={-2,-1.5,-1,-0.5,0,0.5,1,1.5},
        grid=major,
        grid style={dashed, black!30},
        axis background/.style={fill=white},
        tick label style={font=\small},
        label style={font=\small}
    ]

    \addplot[black, thick, solid, samples=200, domain=-1.5:1.5] {cbrt(x)} node[pos=0.3, above left] {$y=\sqrt[3]{x}$};
    \addplot[black, thick, dashed, samples=200, domain=-1.5:0] {x^2-2} node[pos=0.8, below left] {$x=-\sqrt{y+2}$};
    \addplot[black, thick, dotted, samples=200, domain=-0.5:1.2] {3*x-2} node[pos=0.6, below right] {$y=3x-2$};

    \begin{scope}
        \clip plot[samples=200, domain=-1:0] (\x,{cbrt(\x)}) -- 
              plot[samples=200, domain=0:1] (\x,{cbrt(\x)}) --
              (1,1) -- 
              plot[samples=200, domain=1:0] (\x,{3*\x-2}) --
              (0,-2) --
              plot[samples=200, domain=0:-1] (\x,{(\x)^2-2}) --
              cycle;
        \pattern[pattern=north east lines, pattern color=black!60] (-1.5,-2.5) rectangle (1.5,1.5);
    \end{scope}

    \node at (0.2,-0.8) {\large $D$};

  \end{axis}
\end{tikzpicture}
